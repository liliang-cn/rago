# 1. Ingest documents into your local knowledge base
$ rago rag ingest ./my-docs

# 2. Query with semantic search
$ rago rag query "How does the system handle privacy?"

# 3. Run an autonomous agent to perform a task
$ rago agent run "analyze the logs and create a summary report"

# 4. Interactive chat with MCP tool support
$ rago chat
package main

import (
    "github.com/liliang-cn/rago/v2/pkg/agent"
)

func main() {
    svc, _ := agent.New(&agent.AgentConfig{
        Name:         "my-assistant",
        EnableRAG:    true,
        EnableMCP:    true,
        EnableSkills: true,
        EnableRouter: true,
    })
    
    svc.Run(ctx, "What can you help me with?")
}
